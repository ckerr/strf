== Miscellaneous

=== Class template `rank` [[rank]]

[source,cpp,subs=normal]
----
namespace strf {

template <std::size_t N>
struct rank: rank<N - 1>
{
    explicit rank() = default;
};

template <>
struct rank<0>
{
    explicit rank() = default;
};

} // namespace strf
----

=== Class template `tag` [[tag]]

[source,cpp,subs=normal]
----
namespace strf {

template <typename\... >
struct tag
{
    explicit tag() = default;
};

template <typename T>
struct tag<T>
{
    explicit constexpr STRF_HD tag() noexcept { }
    using type = T;
};

} // namespace strf
----
